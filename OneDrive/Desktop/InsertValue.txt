#include <stdio.h>

int main()
{
    printf("Enter no. of elements\n");
    int n;
    scanf("%d",&n);
    int arr[n+1];
    printf("Enter elements");
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int val,ind;
    printf("Enter value to be inserted and at which index");
    scanf("%d %d",&val,&ind);
    for(int i=n-1;i>=ind;i--){
        arr[i+1]=arr[i];
    }
    arr[ind]=val;
    for(int i=0;i<=n;i++){
        printf("%d \t",arr[i]);
    }
}
