#include <stdio.h>
#include <stdlib.h>
struct Student{
    int data;
    struct Student* link;
};
int main()
{
    int n;
    struct Student *prevNode;
    printf("Enter number of nodes to be added:\n");
    scanf("%d",&n);
    struct Student *head;
    head=NULL;
    for(int i=0;i<n;i++){
        if(i==0){
        prevNode = (struct Student*) malloc(sizeof(struct Student));
        prevNode->data=10;
        prevNode->link=NULL;
        head=prevNode;
        }
        else{
          struct Student *nextNode;
          nextNode = (struct Student*) malloc(sizeof(struct Student));
          nextNode->data=10;
          nextNode->link=NULL;
          prevNode->link=nextNode;
          prevNode=nextNode;
         
        }
    }
    struct Student *t = head;
    while(t!=NULL){
        printf("The data is %d\n",t->data);
        t=t->link;
    }
    return 0;
}
