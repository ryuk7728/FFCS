#include<iostream>
#include<cstring>
using namespace std;
class Employee{
    char name[50],dob[15],des[50];
    int salary,ID;
    public:
    Employee(){
        ID=0;
        strcpy(name,"");
        strcpy(dob,"");
        strcpy(des,"");
        salary=0;
        }
    void setData(int ID1,char name1[],int salary1,char dob1[],char des1[]){
        ID=ID1;
        strcpy(name,name1);
        salary=salary1;
        strcpy(dob,dob1);
        strcpy(des,des1);
    }
    static void HighestSalary(Employee arr[],int n){
        int t=-1;
        int maxSal=-1;
        for(int i=0;i<n;i++){
            if(arr[i].salary>maxSal){
                maxSal=arr[i].salary;
                t=i;
            }
        }
        cout<<"Name: "<<arr[t].name;
        
        
    }
};
int main(){
    int n;
    char name[50],dob[15],des[50];
    int salary,ID;
    cin>>n;
    Employee arr[n];
    for(int i=0;i<n;i++){
        cout<<"Enter name\n";
        cin>>name;
        cout<<"Enter salary\n";
        cin>>salary;
        arr[i].setData(ID,name,salary,dob,des);
    }
    Employee::HighestSalary(arr,n);
    return 0;
}
