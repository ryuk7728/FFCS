int top=-1,front=-1,rear=-1,fac=0;
void push(int n,int stk[],int data){
    if(top==(n-1)){
        printf("Stack is Full");
    }
    else{
        top+=1;
        stk[top]=data;
    }
}
int pop(int stk[]){
    if(top==-1){
        printf("Stack is Empty");
    }
    else{
        return stk[top--];
    }
}
